# 作业38：线程同步：[[互斥量]] 和 [[条件变量]]

## 编程题

修改 [[互斥量]] 中的 `thread_incr.c` 程序，以便线程起始函数在每次循环中都能输出 `glob` 的当前值以及能对线程做唯一标识的标识符。可将线程的这一唯一标识指定为创建线程的函数 `pthread_create()` 的调用参数。对于这一程序，需要将线程起始函数的参数改为指针，指向包含线程唯一标识和循环次数限制的数据结构。运行该程序，将输出重定向至一文件，查看内核在调度两线程交替执行时 `glob` 的变化情况。

```c title:theads/thread_incr.c

#include <pthread.h>

#include "base.h"

static int glob = 0;

struct args {
    long loops;
    int id;
};

void *incr(void *arg) {
    const struct args args = *(struct args *)arg;
    for(int i = 0; i < args.loops; i++) {
        int loc = glob;
        loc++;
        glob = loc;
        printf("[Thread %d]: glob = %d\n", args.id, glob);
    }
    return nullptr;
}


int main(int argc, char *argv[]) {
    if(argc != 2 || strcmp(argv[1], "--help") == 0) {
        usageErr("%s loops\n", argv[0]);
    }

    long loops = strtol(argv[1], nullptr, 10);
    pthread_t tids[2];
    int err = pthread_create(tids, nullptr, incr, &(struct args){loops, 0});
    if(err != 0) {
        fprintf(stderr, "pthread_create() failed with error %s\n", strerror(err));
        return -1;
    }
    err = pthread_create(tids+1, nullptr, incr, &(struct args){loops, 1});
    if(err != 0) {
        fprintf(stderr, "pthread_create() failed with error %s\n", strerror(err));
        return -1;
    }
    err = pthread_join(tids[0], nullptr);
    if(err != 0) {
        fprintf(stderr, "pthread_join() failed with error %s\n", strerror(err));
        return -1;
    }
    err = pthread_join(tids[1], nullptr);
    if(err != 0) {
        fprintf(stderr, "pthread_join() failed with error %s\n", strerror(err));
        return -1;
    }
    printf("glob = %d\n", glob);
    return 0;
}
```

编译运行该程序，并将输出重定向到 `tfile` 中

```shell
➜  build ./thread_incr 1000000 > tfile
```

尝试使用 $2$ 个线程对同一个全局变量各加 $2000$ 万，统计加锁解锁的消耗

```c

#include <pthread.h>
#include <sys/time.h>

#include "base.h"

static int glob = 0;

// 声明互斥量，并初始化
static pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

void *incr(void *arg) {
    const int loops = *(int *)arg;
    struct timeval start, end;
    // 获取当前系统时间，高分辨的时间
    gettimeofday(&start, nullptr);
    for(int i = 0; i < loops; i++) {
        // 加锁
        int err = pthread_mutex_lock(&mutex);
        if(err != 0) {
            fprintf(stderr, "pthread_mutex_lock() failed with error %s\n", strerror(err));
            exit(EXIT_FAILURE);
        }
        int loc = glob;
        loc++;
        glob = loc;
        // 解锁
        err = pthread_mutex_unlock(&mutex);
        if(err != 0) {
            fprintf(stderr, "pthread_mutex_unlock() failed with error %s\n", strerror(err));
            exit(EXIT_FAILURE);
        }
    }
    // 获取计算结束后当前时间
    gettimeofday(&end, nullptr);
    printf("Thread[TID %ld] takes %ld us\n", pthread_self(), (end.tv_sec - start.tv_sec) * 1000 * 1000 + (end.tv_usec - start.tv_usec));
    return nullptr;
}

int main(int argc, char *argv[]) {
    if(argc != 2 || strcmp(argv[1], "--help") == 0) {
        usageErr("%s loops\n", argv[0]);
    }

    long loops = strtol(argv[1], nullptr, 10);
    pthread_t tids[2];
    int err = pthread_create(tids, nullptr, incr, &loops);
    if(err != 0) {
        fprintf(stderr, "pthread_create() failed with error %s\n", strerror(err));
        return -1;
    }
    err = pthread_create(tids+1, nullptr, incr, &loops);
    if(err != 0) {
        fprintf(stderr, "pthread_create() failed with error %s\n", strerror(err));
        return -1;
    }
    err = pthread_join(tids[0], nullptr);
    if(err != 0) {
        fprintf(stderr, "pthread_join() failed with error %s\n", strerror(err));
        return -1;
    }
    err = pthread_join(tids[1], nullptr);
    if(err != 0) {
        fprintf(stderr, "pthread_join() failed with error %s\n", strerror(err));
        return -1;
    }
    printf("glob = %d\n", glob);
    return 0;
}
```

使用互斥锁和条件变量保证线程 $B$ 在线程 $A$ 之后运行

```c
#include <signal.h>
#include <pthread.h>
#include <unistd.h>

#include "base.h"

static pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
static pthread_cond_t cond = PTHREAD_COND_INITIALIZER;
static bool flag = false;

void *threadA(void *arg) {
    printf("Before A!\n");
    sleep(3);
    printf("After A\n");

	// 线程 A 要结束了，改变 flag 的状态
    pthread_mutex_lock(&mutex);
    flag = true;
    pthread_mutex_unlock(&mutex);
    // 通知条件变量，阻塞的线程可以执行了
    pthread_cond_signal(&cond);

    return NULL;
}

void *threadB(void *arg) {

	// 检查是否符合条件
   pthread_mutex_lock(&mutex);
    while (!flag) {
        pthread_cond_wait(&cond, &mutex);
    }
    pthread_mutex_unlock(&mutex);
    printf("Before B!\n");
    sleep(3);
    printf("After B\n");
    return NULL;
}

int main(void) {

    pthread_t t1, t2;

    pthread_create(&t1, nullptr, threadA, nullptr);
    pthread_create(&t2, nullptr, threadB, nullptr);

    pthread_join(t1, nullptr);
    pthread_join(t2, nullptr);
    return 0;
}
```

使用信号保证线程 $B$ 在线程 $A$ 之后运行

```c
#include <signal.h>
#include <pthread.h>
#include <unistd.h>

#include "base.h"

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

void *threadA(void *arg) {
    pthread_t thread;
    thread = *(pthread_t *)arg;
    printf("Before A!\n");
    sleep(3);
    printf("After A\n");
    pthread_kill(thread, SIGUSR1);
    return NULL;
}

void *threadB(void *arg) {
    sigset_t unBlocked;
    sigemptyset(&unBlocked);
    sigaddset(&unBlocked, SIGUSR1);
    int sig = 0;
    while (sig != SIGUSR1) {
        sigwait(&unBlocked, &sig);
    }

    printf("Before B!\n");
    sleep(3);
    printf("After B\n");
    return NULL;
}

int main(void) {

    sigset_t blockMask;
    sigfillset(&blockMask);

    sigprocmask(SIG_BLOCK, &blockMask, nullptr);

    pthread_t t1, t2;

    pthread_create(&t2, nullptr, threadB, nullptr);
    pthread_create(&t1, nullptr, threadA, &t2);

    pthread_join(t1, nullptr);
    pthread_join(t2, nullptr);
    return 0;
}
```

### 生产者-消费者模型

在 [[互斥量]] 和 [[条件变量]] 中我们介绍了生产者消费者模型，但是它们被耦合在了一起。这里我们通过 **队列** 缓冲生产者产生的数据，让消费者从队列中读取

```c

#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

typedef struct node {
    int data;
    struct node *next;
}node_t;

node_t * newNode(int data) {
    node_t *node = (node_t *)malloc(sizeof(node_t));
    if(node == nullptr) {
        return nullptr;
    }
    node->data = data;
    node->next = nullptr;
    return node;
}

typedef struct queue {
    node_t *head;
    node_t *tail;
    int size;
    pthread_mutex_t __mutex;
    pthread_cond_t __cond;
} queue_t;


int queue_init(queue_t *queue) {
    if (queue == nullptr) {
        return -1;
    }
    queue->head = nullptr;
    queue->tail = nullptr;
    queue->size = 0;
    pthread_mutex_init(&queue->__mutex, nullptr);
    pthread_cond_init(&queue->__cond, nullptr);
    return 0;
}

int enqueue(int data, queue_t *queue) {
    if(queue == nullptr) {
        return -1;
    }
    node_t *node = newNode(data);
    if(node == nullptr) {
        return -1;
    }
    pthread_mutex_lock(&queue->__mutex);
    if(queue->head == nullptr) {
        queue->head = node;
        queue->tail = node;

    } else {
        queue->tail->next = node;
        queue->tail = node;
    }
    queue->size++;
    pthread_mutex_unlock(&queue->__mutex);
    pthread_cond_signal(&queue->__cond);
    return 0;
}

int dequeue(queue_t *queue, node_t **node) {
    if(queue == nullptr) {
        return -1;
    }
    pthread_mutex_lock(&queue->__mutex);
    if(queue->head == nullptr) {
        pthread_cond_wait(&queue->__cond, &queue->__mutex);
    }
    *node = queue->head;
    queue->head = queue->head->next;
    queue->size--;
    pthread_mutex_unlock(&queue->__mutex);
    return 0;
}

void *producer(void *arg) {
    queue_t *queue = (queue_t *)arg;
    unsigned int seed = time(nullptr);
    while(true) {
        int data = rand_r(&seed) % 100;
        enqueue(data, queue);
        printf("Producer: data=%d\n", data);
        usleep(500 * 1000);
    }
    return NULL;
}

void *consumer(void *arg) {
    queue_t *queue = (queue_t *)arg;
    while(true) {
        node_t *node;
        dequeue(queue, &node);
        printf("consumer: data=%d\n", node->data);
        sleep(1);
    }
    return NULL;
}

void *traverse(void *arg) {
    queue_t *queue = (queue_t *)arg;
    while (true) {
        sleep(3);
        pthread_mutex_lock(&queue->__mutex);
        node_t *node = queue->head;
        printf("traverse: [");
        while(node != nullptr) {
            printf("%-4d", node->data);
            node = node->next;
        }
        printf("]\n");
        pthread_mutex_unlock(&queue->__mutex);
    }
}

int main() {
    setvbuf(stdout, nullptr, _IONBF, 0);
    pthread_t t1;
    pthread_t t2;
    pthread_t t3;
    queue_t queue;
    queue_init(&queue);
    pthread_create(&t1, nullptr, producer, &queue);
    pthread_create(&t2, nullptr, consumer, &queue);
    pthread_create(&t3, nullptr, traverse, &queue);
    pthread_join(t1, nullptr);
    pthread_join(t2, nullptr);
    pthread_join(t3, nullptr);
    return 0;
}
```
